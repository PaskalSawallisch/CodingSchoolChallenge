public class ReverseString { public static void main(String[] args) { // Beispiele String input1 = "codingschool"; String input2 = "mustermann"; System.out.println("Input: '" + input1 + "' - Output: '" + reverseString(input1) + "'"); System.out.println("Input: '" + input2 + "' - Output: '" + reverseString(input2) + "'"); } public static String reverseString(String input) { // StringBuilder verwenden, um den umgekehrten String zu konstruieren StringBuilder reversed = new StringBuilder(); // Durchlaufe das Eingabewort rückwärts und füge jeden Buchstaben zum StringBuilder hinzu for (int i = input.length() - 1; i >= 0; i--) { reversed.append(input.charAt(i)); } // Konvertiere den StringBuilder zurück in einen String und gib ihn zurück return reversed.toString(); } }
